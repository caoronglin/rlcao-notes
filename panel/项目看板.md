const projects = dv.pages('"项目"')
  .where(p => p.status == "doing")
  .sort(p => p.priority, 'asc');

if (projects.length) {
  dv.table(
    ["项目", "优先级", "下次行动"],
    projects.map(p => [p.file.link, p.priority || "—", p.nextAction || "—"])
  );
} else {
  dv.paragraph("✨ 暂无进行中的项目");
}
